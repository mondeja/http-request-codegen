const fetch = require('node-fetch');

fetch(
  'http://localhost:8876',
  {  
    method: 'POST',
    body: JSON.stringify({
      'param-int': '1',
      'param-float': '0.777',
      'param-bool': 'True'
    }),
    headers: {
      'Content-Type': 'application/json'
    }
  }
).then(function(response) {
  console.log(response);
}).catch(function(error) {
  console.error('Error:', error);
});
