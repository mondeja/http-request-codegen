const fetch = require('node-fetch');

fetch(
  'http://localhost:8876',
  {  
    method: 'POST',
    body: new URLSearchParams({
      'param-1': 'value-1'
    }),
    headers: {
      'Content-Type': 'application/x-www-form-urlencoded'
    }
  }
).then(function(response) {
  console.log(response);
}).catch(function(error) {
  console.error('Error:', error);
});
