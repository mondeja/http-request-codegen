const fetch = require('node-fetch');

fetch(
  'http://localhost:8876',
  {  
    method: 'POST',
    body: JSON.stringify({
      'param-1': 'value-1',
      'param-2': '7.77'
    }),
    headers: {
      'Content-Type': 'application/json',
      'Accept-Language': 'fr'
    },
    'timeout': 5
  }
).then(function(response) {
  console.log(response);
}).catch(function(err) {
  console.error('Error:', err);
});
